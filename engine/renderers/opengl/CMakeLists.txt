include_directories(${CMAKE_CURRENT_BINARY_DIR})

message(STATUS "Generator ${CMAKE_GENERATOR}")
if(Windows)
set(QUOTE_START "\"")
endif(Windows)

if(MSVC)
add_custom_command(OUTPUT
                        ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        COMMAND ${SED_EXECUTABLE} "s/^\\s*\\(gl\\S*\\)\\s*$/extern PFN\\U\\1\\EPROC \\1$<SEMICOLON>/g" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        COMMAND ${SED_EXECUTABLE} "s/^\\s*\\(gl\\S*\\)\\s*$/PFN\\U\\1\\EPROC \\1 = nullptr$<SEMICOLON>/g" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        COMMAND ${SED_EXECUTABLE} "s/^\\s*\\(gl\\S*\\)\\s*$/GLGETPROCADDRESS ( PFN\\U\\1\\EPROC, \\1 )$<SEMICOLON>/g" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        COMMAND ${SED_EXECUTABLE} -i "s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g" ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        COMMAND ${SED_EXECUTABLE} -i "s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g" ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        COMMAND ${SED_EXECUTABLE} -i "s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g" ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        DEPENDS
                            ${CMAKE_CURRENT_SOURCE_DIR}/glFunctions.txt
                        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
                        COMMENT "Generating OpenGL functions for ${CMAKE_GENERATOR}.")
elseif(UNIX OR MSYS)
add_custom_command(OUTPUT
                        ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        COMMAND ${SED_EXECUTABLE} "\"s/^\\s*\\(gl\\S*\\)\\s*$$/extern PFN\\U\\1\\EPROC \\1$<SEMICOLON>/g\"" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        COMMAND ${SED_EXECUTABLE} "\"s/^\\s*\\(gl\\S*\\)\\s*$$/PFN\\U\\1\\EPROC \\1 = nullptr$<SEMICOLON>/g\"" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        COMMAND ${SED_EXECUTABLE} "\"s/^\\s*\\(gl\\S*\\)\\s*$$/GLGETPROCADDRESS ( PFN\\U\\1\\EPROC, \\1 )$<SEMICOLON>/g\"" glFunctions.txt > ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        COMMAND ${SED_EXECUTABLE} -i "\"s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g\"" ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
                        COMMAND ${SED_EXECUTABLE} -i "\"s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g\"" ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
                        COMMAND ${SED_EXECUTABLE} -i "\"s/\\*DISCLAIMER\\*/ THIS FILE WAS AUTOGENERATED, DO NOT EDIT, edit glFunctions.txt instead. /g\"" ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
                        DEPENDS
                            ${CMAKE_CURRENT_SOURCE_DIR}/glFunctions.txt
                        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
                        COMMENT "Generating OpenGL functions for ${CMAKE_GENERATOR}.")
endif()

set(OPENGL_RENDERER_HEADERS
    ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
    ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
    ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
    OpenGLRenderer.h
    OpenGLWindow.h
    OpenGLFunctions.h
    OpenGLPipeline.h
    OpenGLMaterial.h
    OpenGLTexture.h
    OpenGLModel.h
    OpenGLSkeleton.h
    OpenGLMesh.h)

if(UNIX)
    set(PLATFORM_DEPENDENT_SOURCES OpenGLRendererX.cpp)
elseif(WIN32)
    set(PLATFORM_DEPENDENT_SOURCES OpenGLRendererWindows.cpp)
endif()

set(OPENGL_RENDERER_SOURCES
    OpenGLRenderer.cpp
    OpenGLWindow.cpp
    OpenGLPipeline.cpp
    OpenGLMaterial.cpp
    OpenGLTexture.cpp
    OpenGLModel.cpp
    OpenGLSkeleton.cpp
    OpenGLMesh.cpp
    OpenGLFunctions.cpp
    ${PLATFORM_DEPENDENT_SOURCES}
    Plugin.cpp)

#set_source_files_properties(OpenGLFunctions.cpp PROPERTIES COMPILE_FLAGS "-E")

add_library(OpenGLRenderer SHARED
    ${OPENGL_RENDERER_HEADERS} ${OPENGL_RENDERER_SOURCES}
    ${CMAKE_SOURCE_DIR}/include/aeongames/Renderer.h)

target_link_libraries(OpenGLRenderer ${OPENGL_LIBRARIES} AeonEngine ProtoBufClasses Threads::Threads)

if(MSVC)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-WX -D_CRT_SECURE_NO_WARNINGS")
elseif(MINGW OR MSYS)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-Wall -Werror -Wfatal-errors" PREFIX "")
elseif(UNIX)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-fpic -Wall  -Werror -Wfatal-errors")
endif()
if(USE_CLANG_TIDY)
    set_target_properties(OpenGLRenderer PROPERTIES
        CXX_CLANG_TIDY "${CLANG_TIDY_EXECUTABLE};-header-filter=aeongames/;${CLANG_TIDY_CHECKS}")
endif()
