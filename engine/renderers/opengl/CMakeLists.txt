include_directories(${CMAKE_CURRENT_BINARY_DIR})

add_custom_command( OUTPUT
                        ${CMAKE_CURRENT_BINARY_DIR}/main.vert.h
                        ${CMAKE_CURRENT_BINARY_DIR}/main.frag.h
                    COMMAND ${XXD_EXECUTABLE} -i main.vert > ${CMAKE_CURRENT_BINARY_DIR}/main.vert.h
                    COMMAND ${XXD_EXECUTABLE} -i main.frag > ${CMAKE_CURRENT_BINARY_DIR}/main.frag.h
                    DEPENDS
                        shaders/main.vert
                        shaders/main.frag
                    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/shaders
                    COMMENT "Generating Shader Headers" VERBATIM)

set(OPENGL_RENDERER_HEADERS
    ${CMAKE_CURRENT_BINARY_DIR}/main.vert.h
    ${CMAKE_CURRENT_BINARY_DIR}/main.frag.h
    ${CMAKE_CURRENT_BINARY_DIR}/glDeclarations.h
    ${CMAKE_CURRENT_BINARY_DIR}/glDefinitions.h
    ${CMAKE_CURRENT_BINARY_DIR}/glAssignments.h
    OpenGLRenderer.h
    OpenGLFunctions.h
    OpenGLProgram.h
    OpenGLMaterial.h
    OpenGLTexture.h
    OpenGLMesh.h
    Uniform.h)

set(OPENGL_RENDERER_SOURCES
    OpenGLRenderer.cpp
    OpenGLProgram.cpp
    OpenGLMaterial.cpp
    OpenGLTexture.cpp
    OpenGLMesh.cpp
    OpenGLFunctions.cpp
    Uniform.cpp
    Plugin.cpp)

source_group("opengl" FILES ${OPENGL_RENDERER_HEADERS} ${OPENGL_RENDERER_SOURCES})

add_library(OpenGLRenderer SHARED
    ${OPENGL_RENDERER_HEADERS} ${OPENGL_RENDERER_SOURCES}
    ${CMAKE_SOURCE_DIR}/include/aeongames/Renderer.h)

target_link_libraries(OpenGLRenderer ${OPENGL_LIBRARIES} AeonEngine ProtoBufClasses Threads::Threads)

if(MSVC)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-WX -D_CRT_SECURE_NO_WARNINGS -DPROTOBUF_USE_DLLS")
elseif(MINGW OR MSYS)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-Wall -Werror -Wfatal-errors -DPROTOBUF_USE_DLLS")
elseif(UNIX)
    set_target_properties(OpenGLRenderer PROPERTIES
        COMPILE_FLAGS "-fpic -Wall  -Werror -Wfatal-errors")
endif()
