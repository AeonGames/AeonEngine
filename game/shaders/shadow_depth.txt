vert: "#version 450\n"
"\n"
"// Directional shadow depth pass. Rasterizes static geometry into the shadow\n"
"// map using the light's world-space view-projection. Mirrors the vertex inputs\n"
"// of cluster_mark.vert / diffuse_map_phong_no_skeleton.vert so the same static\n"
"// meshes bind unchanged. Skinned meshes fall back to their rest pose here.\n"
"#ifdef VULKAN\n"
"layout(set = 8, binding = 0, std140)\n"
"#else\n"
"layout(binding = 5, std140)\n"
"#endif\n"
"uniform ShadowParams\n"
"{\n"
"      mat4 light_view_projection; // world -> light clip space\n"
"      vec4 shadow_params;         // (texel_size, depth_bias, pcf_radius, enabled)\n"
"};\n"
"\n"
"// Per-object model matrices, shared with the uber-pipeline. Indexed by the\n"
"// instance id so single and instanced draws bind unchanged.\n"
"#ifdef VULKAN\n"
"layout(set = 3, binding = 0, std430)\n"
"#else\n"
"layout(binding = 3, std430)\n"
"#endif\n"
"readonly buffer InstanceMatrices\n"
"{\n"
"      mat4 InstanceModelMatrices[];\n"
"};\n"
"#ifdef VULKAN\n"
"#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceIndex]\n"
"#else\n"
"#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceID]\n"
"#endif\n"
"\n"
"layout(location = 0) in vec3 VertexPosition;\n"
"layout(location = 1) in vec3 VertexNormal;\n"
"layout(location = 2) in vec3 VertexTangent;\n"
"layout(location = 3) in vec3 VertexBitangent;\n"
"layout(location = 4) in vec2 VertexUV;\n"
"\n"
"void main()\n"
"{\n"
"      gl_Position = light_view_projection * MODEL_MATRIX * vec4 ( VertexPosition, 1.0 );\n"
"#ifdef VULKAN\n"
"      // Matrix4x4::Ortho produces GL-style NDC z in [-1,1]; Vulkan clips to\n"
"      // [0,1]. Without this remap the near half of the light frustum would be\n"
"      // clipped out of the shadow map. (The perspective camera path tolerates\n"
"      // the raw matrix because its clipped near sliver is tiny; an orthographic\n"
"      // light projection would otherwise lose half its depth range.)\n"
"      gl_Position.z = ( gl_Position.z + gl_Position.w ) * 0.5;\n"
"#endif\n"
"}\n"
frag: "#version 450\n"
"\n"
"// Directional shadow depth pass fragment shader. Only depth matters: the depth\n"
"// attachment is the shadow map. The Vulkan render pass carries a throwaway\n"
"// colour attachment (DONT_CARE) for render-pass compatibility with the main\n"
"// pipelines; the OpenGL shadow framebuffer has no colour attachment at all\n"
"// (glDrawBuffer(GL_NONE)). The colour value is discarded, but the ALPHA must be\n"
"// 1.0: the shared pipeline enables alpha-to-coverage, so an alpha of 0 would\n"
"// produce zero coverage and kill the fragment before its depth is written,\n"
"// leaving the shadow map empty on Vulkan.\n"
"layout(location = 0) out vec4 FragColor;\n"
"\n"
"void main()\n"
"{\n"
"      FragColor = vec4 ( 1.0 );\n"
"}\n"
