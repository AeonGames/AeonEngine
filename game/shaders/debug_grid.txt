vert:
  "#version 450\n"
  "\n"
  "// Analytic infinite ground grid for the renderer's debug-geometry step.\n"
  "// A single full-screen triangle is unprojected to world space; the\n"
  "// fragment stage intersects the eye ray with the z = 0 ground plane\n"
  "// (this engine is +X right, +Y forward, +Z up) and draws the grid\n"
  "// analytically, so the grid is truly infinite and needs no per-cell\n"
  "// geometry. Shares the Proj/View Matrices UBO with every other pipeline.\n"
  "#ifdef VULKAN\n"
  "layout(set = 0, binding = 0, std140)\n"
  "#else\n"
  "layout(binding = 0, std140)\n"
  "#endif\n"
  "uniform Matrices\n"
  "{\n"
  "      mat4 ProjectionMatrix;\n"
  "      mat4 ViewMatrix;\n"
  "};\n"
  "\n"
  "layout(location = 0) in vec3 VertexPosition;\n"
  "layout(location = 0) out vec3 NearPoint;\n"
  "layout(location = 1) out vec3 FarPoint;\n"
  "layout(location = 2) flat out vec3 CameraPosition;\n"
  "\n"
  "vec3 Unproject ( vec2 aClipXY, float aClipZ, mat4 aInvViewProjection )\n"
  "{\n"
  "      vec4 world = aInvViewProjection * vec4 ( aClipXY, aClipZ, 1.0 );\n"
  "      return world.xyz / world.w;\n"
  "}\n"
  "\n"
  "void main()\n"
  "{\n"
  "      mat4 inv_view_projection = inverse ( ProjectionMatrix * ViewMatrix );\n"
  "      vec2 clip = VertexPosition.xy;\n"
  "      // The fullscreen triangle is authored counter-clockwise in OpenGL's\n"
  "      // Y-up window space. Vulkan's window space is Y-down, which flips the\n"
  "      // winding to clockwise and gets it back-face culled, so mirror Y there.\n"
  "      // gl_Position and the unprojected ray both use this same clip value, so\n"
  "      // each fragment still resolves the world point under its own pixel.\n"
  "#ifdef VULKAN\n"
  "      clip.y = -clip.y;\n"
  "#endif\n"
  "      // Unproject the same clip XY at both depth extremes of this engine's\n"
  "      // [-1, 1] projection to recover the eye ray in world space.\n"
  "      NearPoint = Unproject ( clip, -1.0, inv_view_projection );\n"
  "      FarPoint = Unproject ( clip, 1.0, inv_view_projection );\n"
  "      CameraPosition = ( inverse ( ViewMatrix ) * vec4 ( 0.0, 0.0, 0.0, 1.0 ) ).xyz;\n"
  "      gl_Position = vec4 ( clip, 0.0, 1.0 );\n"
  "}\n"

frag:
  "#version 450\n"
  "\n"
  "layout(location = 0) in vec3 NearPoint;\n"
  "layout(location = 1) in vec3 FarPoint;\n"
  "layout(location = 2) flat in vec3 CameraPosition;\n"
  "layout(location = 0) out vec4 FragColor;\n"
  "\n"
  "#ifdef VULKAN\n"
  "layout(set = 0, binding = 0, std140)\n"
  "#else\n"
  "layout(binding = 0, std140)\n"
  "#endif\n"
  "uniform Matrices\n"
  "{\n"
  "      mat4 ProjectionMatrix;\n"
  "      mat4 ViewMatrix;\n"
  "};\n"
  "\n"
  "#ifdef VULKAN\n"
  "layout(set = 1, binding = 0, std140)\n"
  "#else\n"
  "layout(binding = 1, std140)\n"
  "#endif\n"
  "uniform Material\n"
  "{\n"
  "      vec4 GridColor;\n"
  "      vec4 GridMajorColor;\n"
  "      vec4 AxisXColor;\n"
  "      vec4 AxisYColor;\n"
  "      float CellSize;\n"
  "      float MajorInterval;\n"
  "      float FadeDistance;\n"
  "};\n"
  "\n"
  "// Distance, in pixels, from a coordinate to the nearest integer grid line\n"
  "// (one unit = one cell). Uses screen-space derivatives so the measure is\n"
  "// resolution independent.\n"
  "float GridLineDistancePixels ( vec2 aCoord )\n"
  "{\n"
  "      vec2 derivative = fwidth ( aCoord );\n"
  "      vec2 distance = abs ( fract ( aCoord - 0.5 ) - 0.5 ) / derivative;\n"
  "      return min ( distance.x, distance.y );\n"
  "}\n"
  "\n"
  "void main()\n"
  "{\n"
  "      // Intersect the eye ray with the z = 0 ground plane.\n"
  "      float t = -NearPoint.z / ( FarPoint.z - NearPoint.z );\n"
  "      if ( t <= 0.0 )\n"
  "      {\n"
  "          discard;\n"
  "      }\n"
  "      vec3 world = NearPoint + t * ( FarPoint - NearPoint );\n"
  "\n"
  "      // Fade the grid out with horizontal distance from the camera so the\n"
  "      // far field does not turn into a solid moire sheet.\n"
  "      float camera_distance = length ( world.xy - CameraPosition.xy );\n"
  "      if ( camera_distance >= FadeDistance )\n"
  "      {\n"
  "          discard;\n"
  "      }\n"
  "\n"
  "      // Pick the highest-priority line covering this pixel: world axes first,\n"
  "      // then major lines, then minor lines. Each is a crisp solid line of a\n"
  "      // fixed pixel half-width; the pipeline is single-sample with no alpha\n"
  "      // blending (Vulkan also enables alpha-to-coverage), so a fractional\n"
  "      // alpha would dither into a stipple. Comparing pixel distances against\n"
  "      // a half-width and emitting opaque pixels keeps every line uniform.\n"
  "      const float minor_half_width = 0.5;\n"
  "      const float major_half_width = 0.75;\n"
  "      const float axis_half_width  = 1.0;\n"
  "\n"
  "      vec2 pixel = fwidth ( world.xy );\n"
  "      float x_axis_dist = abs ( world.y ) / pixel.y;\n"
  "      float y_axis_dist = abs ( world.x ) / pixel.x;\n"
  "      float major_dist = GridLineDistancePixels ( world.xy / ( CellSize * MajorInterval ) );\n"
  "      float minor_dist = GridLineDistancePixels ( world.xy / CellSize );\n"
  "\n"
  "      vec3 color;\n"
  "      if ( x_axis_dist <= axis_half_width )\n"
  "      {\n"
  "          color = AxisXColor.rgb;\n"
  "      }\n"
  "      else if ( y_axis_dist <= axis_half_width )\n"
  "      {\n"
  "          color = AxisYColor.rgb;\n"
  "      }\n"
  "      else if ( major_dist <= major_half_width )\n"
  "      {\n"
  "          color = GridMajorColor.rgb;\n"
  "      }\n"
  "      else if ( minor_dist <= minor_half_width )\n"
  "      {\n"
  "          color = GridColor.rgb;\n"
  "      }\n"
  "      else\n"
  "      {\n"
  "          discard;\n"
  "      }\n"
  "      FragColor = vec4 ( color, 1.0 );\n"
  "\n"
  "      // Write true scene depth so solid geometry occludes the grid. This\n"
  "      // engine's projection emits [-1, 1] clip depth on both backends;\n"
  "      // OpenGL maps that to the [0, 1] depth range, Vulkan uses it directly.\n"
  "      vec4 clip = ProjectionMatrix * ViewMatrix * vec4 ( world, 1.0 );\n"
  "      float ndc_depth = clip.z / clip.w;\n"
  "#ifdef VULKAN\n"
  "      gl_FragDepth = ndc_depth;\n"
  "#else\n"
  "      gl_FragDepth = ndc_depth * 0.5 + 0.5;\n"
  "#endif\n"
  "}\n"
topology_class: TRIANGLE
