vert: "#version 450\n"
"\n"
"// Point shadow depth pass (single-pass, geometry-shader variant). Transforms\n"
"// each vertex to world space and forwards it; the projection into each of the\n"
"// six cube faces happens in point_shadow_depth.geom, which replicates the\n"
"// primitive across the six cube-map-array layers in one draw. The fragment\n"
"// shader writes normalized radial distance from the light (view-independent),\n"
"// so no per-face projection happens here.\n"
"//\n"
"// NOTE: a future optimization is to drop the geometry shader on Vulkan and use\n"
"// multiview (gl_ViewIndex) instead, projecting per view in this vertex shader.\n"
"// That requires per-renderer pipeline variants (no geometry stage on the Vulkan\n"
"// build), so for now both backends share this geometry-shader path.\n"
"\n"
"// Per-object model matrices, shared with the uber-pipeline. Indexed by the\n"
"// instance id so single and instanced draws bind unchanged.\n"
"#ifdef VULKAN\n"
"layout(set = 3, binding = 0, std430)\n"
"#else\n"
"layout(binding = 3, std430)\n"
"#endif\n"
"readonly buffer InstanceMatrices\n"
"{\n"
"      mat4 InstanceModelMatrices[];\n"
"};\n"
"#ifdef VULKAN\n"
"#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceIndex]\n"
"#else\n"
"#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceID]\n"
"#endif\n"
"\n"
"layout(location = 0) in vec3 VertexPosition;\n"
"layout(location = 1) in vec3 VertexNormal;\n"
"layout(location = 2) in vec3 VertexTangent;\n"
"layout(location = 3) in vec3 VertexBitangent;\n"
"layout(location = 4) in vec2 VertexUV;\n"
"\n"
"layout(location = 0) out vec3 vWorldPosition;\n"
"\n"
"void main()\n"
"{\n"
"      vWorldPosition = ( MODEL_MATRIX * vec4 ( VertexPosition, 1.0 ) ).xyz;\n"
"}\n"
frag: "#version 450\n"
"\n"
"// Point shadow depth pass fragment shader. Writes the normalized radial\n"
"// distance from the light to this fragment as the depth value, so the shading\n"
"// pass can compare against the same metric (length(frag - light) / radius)\n"
"// regardless of which cube face stored the texel. Writing gl_FragDepth makes\n"
"// the comparison view-independent and backend-independent (no GL/Vulkan NDC\n"
"// remap), at the cost of early-Z; that is fine for a depth-only shadow pass.\n"
"//\n"
"// The colour attachment is a throwaway carried only for render-pass\n"
"// compatibility with the main pipelines (see shadow_depth.frag); its alpha must\n"
"// be 1.0 because the shared pipeline enables alpha-to-coverage.\n"
"#ifdef VULKAN\n"
"layout(set = 8, binding = 0, std140)\n"
"#else\n"
"layout(binding = 5, std140)\n"
"#endif\n"
"uniform ShadowParams\n"
"{\n"
"      mat4 face_view_projection[6]; // POINT_SHADOW_FACES\n"
"      vec4 light_position_radius;   // xyz world pos, w radius\n"
"      vec4 face_params;             // x = base cube-array layer\n"
"};\n"
"\n"
"layout(location = 0) in vec3 gWorldPosition;\n"
"layout(location = 0) out vec4 FragColor;\n"
"\n"
"void main()\n"
"{\n"
"      float dist = length ( gWorldPosition - light_position_radius.xyz );\n"
"      gl_FragDepth = clamp ( dist / light_position_radius.w, 0.0, 1.0 );\n"
"      FragColor = vec4 ( 1.0 );\n"
"}\n"
geom: "#version 450\n"
"\n"
"// Point shadow depth pass geometry shader (single-pass cube rendering). For each\n"
"// input triangle, emits it six times -- once per cube face -- routing each copy\n"
"// to its cube-map-array layer with gl_Layer and projecting it with that face's\n"
"// view-projection. This replaces the old six-draws-per-caster loop with a single\n"
"// draw per caster.\n"
"//\n"
"// The block is named \"ShadowParams\" so it binds through the same engine slot the\n"
"// directional/spot depth pass uses (its layout differs per pipeline; reflection\n"
"// is per-pipeline). gl_Layer = base_layer + face: base_layer is caster*6 on\n"
"// OpenGL (the whole cube-map array is attached, so layers are absolute) and 0 on\n"
"// Vulkan (a per-caster six-layer framebuffer is bound, so layers are relative).\n"
"//\n"
"// NOTE: a future Vulkan-multiview path would remove this geometry stage entirely\n"
"// (replicating per gl_ViewIndex in hardware); see point_shadow_depth.vert.\n"
"layout ( triangles ) in;\n"
"layout ( triangle_strip, max_vertices = 18 ) out;\n"
"\n"
"#ifdef VULKAN\n"
"layout(set = 8, binding = 0, std140)\n"
"#else\n"
"layout(binding = 5, std140)\n"
"#endif\n"
"uniform ShadowParams\n"
"{\n"
"      mat4 face_view_projection[6]; // POINT_SHADOW_FACES\n"
"      vec4 light_position_radius;   // xyz world pos, w radius\n"
"      vec4 face_params;             // x = base cube-array layer\n"
"};\n"
"\n"
"layout(location = 0) in vec3 vWorldPosition[];\n"
"layout(location = 0) out vec3 gWorldPosition;\n"
"\n"
"void main()\n"
"{\n"
"      int base_layer = int ( face_params.x );\n"
"      for ( int face = 0; face < 6; ++face )\n"
"      {\n"
"            gl_Layer = base_layer + face;\n"
"            for ( int i = 0; i < 3; ++i )\n"
"            {\n"
"                  gWorldPosition = vWorldPosition[i];\n"
"                  vec4 clip = face_view_projection[face] * vec4 ( vWorldPosition[i], 1.0 );\n"
"#ifdef VULKAN\n"
"                  // Matrix4x4::Perspective produces GL-style NDC z in [-1,1];\n"
"                  // Vulkan clips to [0,1]. Remap so the near half of the face\n"
"                  // frustum is not clipped (only affects clipping; the stored\n"
"                  // depth is the linear distance written by the fragment shader).\n"
"                  clip.z = ( clip.z + clip.w ) * 0.5;\n"
"#endif\n"
"                  gl_Position = clip;\n"
"                  EmitVertex();\n"
"            }\n"
"            EndPrimitive();\n"
"      }\n"
"}\n"
