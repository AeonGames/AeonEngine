vert {
  key: ""
  value: "#version 450\n\n// Point shadow depth pass. Rasterizes static geometry into one cube-face layer\n// of the point shadow map. Unlike the directional/spot depth pass\n// (shadow_depth.vert), which lets the hardware store projected NDC depth, this\n// pass forwards the fragment\'s world position so the fragment shader can write\n// LINEAR radial distance from the light. That makes the stored value\n// view-independent and identical across all six faces, which the shading pass\n// then compares against length(frag - light) / radius.\n#ifdef VULKAN\nlayout(set = 8, binding = 0, std140)\n#else\nlayout(binding = 5, std140)\n#endif\nuniform ShadowParams\n{\n      mat4 light_view_projection; // world -> this face\'s light clip space\n      vec4 shadow_params;         // (light_x, light_y, light_z, radius)\n};\n\n// Per-object model matrices, shared with the uber-pipeline. Indexed by the\n// instance id so single and instanced draws bind unchanged.\n#ifdef VULKAN\nlayout(set = 3, binding = 0, std430)\n#else\nlayout(binding = 3, std430)\n#endif\nreadonly buffer InstanceMatrices\n{\n      mat4 InstanceModelMatrices[];\n};\n#ifdef VULKAN\n#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceIndex]\n#else\n#define MODEL_MATRIX InstanceModelMatrices[gl_InstanceID]\n#endif\n\nlayout(location = 0) in vec3 VertexPosition;\nlayout(location = 1) in vec3 VertexNormal;\nlayout(location = 2) in vec3 VertexTangent;\nlayout(location = 3) in vec3 VertexBitangent;\nlayout(location = 4) in vec2 VertexUV;\n\nlayout(location = 0) out vec3 world_position;\n\nvoid main()\n{\n      vec4 world = MODEL_MATRIX * vec4 ( VertexPosition, 1.0 );\n      world_position = world.xyz;\n      gl_Position = light_view_projection * world;\n#ifdef VULKAN\n      // Matrix4x4::Perspective produces GL-style NDC z in [-1,1]; Vulkan clips\n      // to [0,1]. Remap so the near half of the face frustum is not clipped.\n      // (Only affects rasterization/clipping here; the stored depth value is\n      // the linear distance written by the fragment shader, not this z.)\n      gl_Position.z = ( gl_Position.z + gl_Position.w ) * 0.5;\n#endif\n}\n"
}
frag {
  key: ""
  value: "#version 450\n\n// Point shadow depth pass fragment shader. Writes the normalized radial\n// distance from the light to this fragment as the depth value, so the shading\n// pass can compare against the same metric (length(frag - light) / radius)\n// regardless of which cube face stored the texel. Writing gl_FragDepth makes\n// the comparison view-independent and backend-independent (no GL/Vulkan NDC\n// remap), at the cost of early-Z; that is fine for a depth-only shadow pass.\n//\n// The colour attachment is a throwaway carried only for render-pass\n// compatibility with the main pipelines (see shadow_depth.frag); its alpha must\n// be 1.0 because the shared pipeline enables alpha-to-coverage.\n#ifdef VULKAN\nlayout(set = 8, binding = 0, std140)\n#else\nlayout(binding = 5, std140)\n#endif\nuniform ShadowParams\n{\n      mat4 light_view_projection;\n      vec4 shadow_params; // (light_x, light_y, light_z, radius)\n};\n\nlayout(location = 0) in vec3 world_position;\nlayout(location = 0) out vec4 FragColor;\n\nvoid main()\n{\n      float dist = length ( world_position - shadow_params.xyz );\n      gl_FragDepth = clamp ( dist / shadow_params.w, 0.0, 1.0 );\n      FragColor = vec4 ( 1.0 );\n}\n"
}
